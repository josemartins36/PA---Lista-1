#include <stdio.h>
#include <stdlib.h>

void ordena_crescente(float *valores, int n) {
    float temp;
    int i, j;
    for (i = 0; i < n; i++) {
        for (j = i + 1; j < n; j++) {
            if (valores[i] > valores[j]) {
                temp = valores[i];
                valores[i] = valores[j];
                valores[j] = temp;
            }
        }
    }
}

int main() {
    int n, i;
    float *valores;
    
    // ler n valores
    printf("Digite o número de valores: ");
    scanf("%d", &n);

    // alocação dinâmica de memória
    valores = (float *)malloc(n * sizeof(float)); 

    // ler os n valores
    printf("Digite os valores:\n");
    for (i = 0; i < n; i++) {
        scanf("%f", &valores[i]);
    }
    
    // ordena esses valores em ordem crescente
    ordena_crescente(valores, n);

    // printar os valores, já ordenados
    printf("Valores em ordem crescente:\n");
    for (i = 0; i < n; i++) {
        printf("%.2f\n", valores[i]);
    }
    
    // libera a memória
    free(valores); 
    return 0;
}
